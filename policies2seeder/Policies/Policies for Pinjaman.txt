#2. Kuarters - Permohonan Menduduki Kuarters.
		#1. Permohonan Menduduki Kuarters Permohonan Masuk Kakitangan LKIM
			#1. Kakitangan 
				#1. Function Listing
					1. "Kakitangan" "view" and search list of application that belongs to themselves.
	
				#2. Function Add Pemohonan 
					1. "Kakitangan" "add" application details.
				   
				#3. Function Papar Butiran
					1. "Kakitangan" "view" application details of selected application. 
					2. "Kakitangan" "view" status of the application updated by "US Pinjaman" and "PN" .
					
			#2. Urus Setia Peringkat Negeri 
				#1. Function Listing 
					1. "US Pinjaman" "view" and search list of application.
					
				#2. Function Papar Butiran
					1. "US Pinjaman" "view" selected application details and its supported documents. 
					2. "US Pinjaman" "update" status to "SEMAK/TIDAK SEMAK" to  the verification of the application.
					3. "US Pinjaman" "update" status to "LULUS/TIDAK LULUS" to  the approval of the application.
					
					IF "TIDAK LULUS", the application is end there.
					IF "LULUS", the application will proceed to the next level.
					
					4. "US Pinjaman" "download" "Surat Tawaran menduduki Kuarters".
					
			#3. Pengarah Negeri 
				#1. Function Listing
					1. "PN" "view" and search list of application for quarters.
					
				#2. Function Papar Butiran
					1. "PN" "view" application details and supported document.
					2. "PN" "view" the status if the application is verified, and approved status by the 'US Pinjaman'.
					3. "PN" "update" status of application approval process.
					
					IF "TIDAK LULUS", the application is end there.
					IF "LULUS", the application will proceed to the next level.
					
			#4. Unit Pengurusan Fasiliti 
				#1. Function Listing
					1. "UPF" "view" and search list of application for quarters.
				#2. Funtion Papar Butiran 
					1. "UPF" "view" the application details and the supported documents.
					2. "UPF" "view" status of the application. 
					3. "UPF" "update" details "Maklumat Pemotongan Mengikut Jenis Kelayakan Kuarters".
					4. "UPF" "download"  "Surat Tawaran menduduki Kuarters".
			
			#5. System 
				1. System generates
					1. Surat Tawaran menduduki Kuarters
					2. Borang Akuan Masuk Rumah
					3. Borang Pemeriksaan Masuk Rumah
					4. Surat Akuan Patuh Kepada Peraturan
					5. Garis Bantuan
				2. System notify the applicant.
					
		#2.	Pemohonan Menduduki Kuarters Permohonan Masuk Bukan Kakitangan LKIM 
			#1. Urus Setia Peringkat Negeri

				#2. Function Papar Butiran

					2. "PN" "update" approval status for the application.
					
					IF "TIDAK LULUS", the application is end there.

			#3. Unit Pengurusan Fasiliti

#3. Kuarters - Permohonan Keluar Kuarters.
		#1. Permohonan Keluar Kakitangan LKIM 
			#1. Kakitangan
				#1. Function Listing
					1."Kakitangan" "view" and search list of application for quarters that belongs to themselves.
				#2. Function Add Permohonan
					1. "Kakitangan" "add" application details and supporting documents.
					
				#3. Function Papar Butiran
					1. "Kakitangan" "view" application details. 
					2. "Kakitangan" "view" the status of verification from 'US Pinjaman'.
					3. "Kakitangan" "view" the status of approval from 'PN'.
					4. "Kakitangan" "download" memo surat and form from the system.
					
			#2. Urus Setia Peringkat Negeri
				#1. Function Listing
					1. "US Pinjaman" "view" and search list of application for 'Keluar Kuarters'.
					
				#2. Function Papar Butiran
					1. "US Pinjaman" "view" the application details and supporting documents for selected application.
					2. "US Pinjaman" "update" the status of verification of the application.
					3. "US Pinjaman" "view" the status of approval from 'PN'.
					4. "US Pinjaman" "update" the all the details related to quarters.
					
			#3. Pengarah Negeri 
				#1. Function Listing
					1. "PN" "view" and search list of application for 'Keluar Kuarters'.
					
				#2. Function Papar Butiran
					1. "PN" "view" the application details and supportings documents for selected application.
					2. "PN" "view" the verification status of the application. 
					3. "PN" "update" the status of approval of the application to either 'LULUS/TIDAK LULUS'.
					
					IF "TIDAK LULUS", the application end there.
					IF "LULUS", the application proceed to next level.
						
			#4. Unit Pengurusan Fasiliti
				#1. Function Listing
					1. "UPF" "view" and search list of application for 'Keluar Kuarters'.
					
				#2. Function Papar Butiran
					1."UPF" "view" the application details related to 'Keluar Kuarters'.
					2."UPF" downlaod memo, surat and borang from the system.
					
					
			#5. System
				1. System generated "Memo Keluar Kuarters", "Borang Akuan Keluar Kuartes",
									"Surat Maklumat dari Negeri"
				2. System notified the applicant involved.
		#2. Permohonan Keluar (Bukan LKIM)
			#1. Urus Setia Peringkat Negeri
				#1. Function Listing 
					1. "US Pinjaman" "view" and search list of application for 'Keluar Kuarters'.
				#2. Function Add Permohonan
					1. "US Pinjaman" "add" application detail for 'Keluar Kuarters'.
					
					IF isArrears (Tunggakan) ==TRUE,
					2. "US Pinjaman" "download" order letter (Surat Arahan Bayaran Tungggakan).
					
					IF isArrers === FALSE, will proceed to next level.
					3. "US Pinjaman" "update" the selected application details.
					
				#3. Function Papar Butiran
			#2. Pengarah Negeri
				#1. Function Listing
					1. "PN" "view" and search list of application for 'Keluar Kuarters'.
				
				#2. Function Papar Butiran 
					1. "PN" "view" the application details for selected application.
					2. "PN" "view" isArrear status whether is settled or not.
					3. "PN" "update" the approval status of the application status.
					
					IF approvalStatus =="TIDAK LULUS", the application is ended.
					IF approvalStatus == "LULUS", the application is proceed to next level.
					
			#3. System 
				IF isArrears == TRUE
				1. System generate order letter. 
				IF approvalStatus=="LULUS"
				2. System generate "borang akuan keluar kuarters", "borang pemeriksaan keluar kuarters",
						"surat makluman dari negeri"
				3. System notified the applicant.
